## 유효한 팰린드롬

```
function solution(s) {
    let answer = "YES";
    s = s.toLowerCase().replace(/[^a-z]/g, "");
    if (s.split("").reverse().join("") !== s) return "NO";
     return answer;
}
let str = "found7, time: study; Yduts; emit, 7Dnuof";
console.log(solution(str));
```

## 숫자만 추출

```
function solution(s) {
    let answer;
    answer = s.replace(/[^0-9]/g, "");
    return parseInt(answer);
}
let str = "g0en2T0s8eSoft";
console.log(solution(str));
```
