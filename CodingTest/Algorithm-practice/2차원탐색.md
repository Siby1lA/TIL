# 2. 2차원 탐색

## 큰 수 출력하기

```
function solution(arr) {
    let answer = [];
    answer.push(arr[0]);
    for (let i = 1; i < arr.length; i++) {
        if (arr[i - 1] < arr[i]) answer.push(arr[i]);
    }
    return answer;
}
let arr = [7, 3, 9, 5, 6, 12];
console.log(solution(arr));
```

## 보이는 학생

```
function solution(arr) {
    let answer = 1,
        max = arr[0];
    for (let i = 1; i < arr.length; i++) {
        if (max < arr[i]) {
        answer++;
        max = arr[i];
        }
    }
    return answer;
}
let arr = [130, 135, 148, 140, 145, 150, 150, 153];
console.log(solution(arr));
```
