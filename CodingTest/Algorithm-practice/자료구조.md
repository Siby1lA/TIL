# 6. 자료구조(스택, 큐)

## 올바른 괄호

### 내 풀이

```
function solution(str) {
    let answer = "YES";
    let stack = [];
    for (let i = 0; i < str.length; i++) {
        if (str[i] === "(") stack.push(str[i]);
        else stack.pop();
    }
    if (stack.length !== 0) answer = "NO";
    return answer;
}
let str = "(()(()))(()";
console.log(solution(str));
```

## 괄호문자제거

### 내 풀이

```
function solution(str) {
    let answer;
    let stack = [];
    for (let i = 0; i < str.length; i++) {
        if (str[i] !== ")") stack.push(str[i]);
        else if (str[i] === ")") {
        // 이쪽부분 틀림
        if (str[i] !== "(") stack.pop();
        // 정답
        while (stack.pop() !== "(");
        }
    }
    answer = stack.join("");
    return answer;
}
let str = "(A(BC)D)EF(G(H)(IJ)K)LM(N)";
console.log(solution(str));
```

### 문제 풀이

```
function solution(s){
    let answer;
    let stack=[];
    for(let x of s){
        if(x===')'){
            while(stack.pop()!=='(');
        }
        else stack.push(x);
    }
    answer=stack.join('');
    return answer;
}
let str="(A(BC)D)EF(G(H)(IJ)K)LM(N)";
console.log(solution(str));
```
